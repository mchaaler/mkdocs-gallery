49 def parse_source_file(file: Path):
50 """Parse source file into AST node.
51  
52 Parameters
53 ----------
54 file : Path
55 File path
56  
57 Returns
58 -------
59 node : AST node
60 content : utf-8 encoded string
61 """
62 # with codecs.open(filename, 'r', 'utf-8') as fid:
63 # content = fid.read()
64 content = file.read_text(encoding="utf-8")

66 # change from Windows format to UNIX for uniformity

71 return node, content
72 except SyntaxError:

141 def extract_file_config(content):
142 """
143 Pull out the file-specific config specified in the docstring.
144 """
145 file_conf = {}
146 for match in re.finditer(INFILE_CONFIG_PATTERN, content):
147 name = match.group(1)
148 value = match.group(3)
149 if value is None: # a flag rather than a config setting
150 continue
151 try:
152 value = ast.literal_eval(value)
153 except (SyntaxError, ValueError):
154 logger.warning(
155 'mkdocs-gallery option %s was passed invalid value %s',
156 name, value)
157 else:
158 file_conf[name] = value
159 return file_conf

162 def split_code_and_text_blocks(source_file: Path, return_node=False) -> Union[Tuple[Dict, List], Tuple[Dict, List, ast.AST]]:
163 """Return list with source file separated into code and text blocks.
164  
165 Parameters
166 ----------
167 source_file : Path
168 Path to the source file.
169 return_node : bool
170 If True, return the ast node.
171  
172 Returns
173 -------
174 file_conf : dict
175 File-specific settings given in source file comments as:
176 ``# mkdocs_gallery_<name> = <value>``
177 blocks : list
178 (label, content, line_number)
179 List where each element is a tuple with the label ('text' or 'code'),
180 the corresponding content string of block and the leading line number
181 node : ast Node
182 The parsed node.
183 """
184 docstring, rest_of_content, lineno, node = _get_docstring_and_rest(source_file)
185 blocks = [('text', docstring, 1)]
186  
187 file_conf = extract_file_config(rest_of_content)
188  
189 pattern = re.compile(r'(?P<header_line>^#{20,}.*|^# ?%%.*)\s(?P<text_content>(?:^#.*\s?)*)', flags=re.M)
190 sub_pat = re.compile('^#', flags=re.M)
191  
192 pos_so_far = 0
193 for match in re.finditer(pattern, rest_of_content):
194 code_block_content = rest_of_content[pos_so_far:match.start()]
195 if code_block_content.strip():
196 blocks.append(('code', code_block_content, lineno))
197 lineno += code_block_content.count('\n')
198  
199 lineno += 1 # Ignored header line of hashes.
200 text_content = match.group('text_content')
201 text_block_content = dedent(re.sub(sub_pat, '', text_content)).lstrip()
202 if text_block_content.strip():
203 blocks.append(('text', text_block_content, lineno))
204 lineno += text_content.count('\n')
205  
206 pos_so_far = match.end()
207  
208 remaining_content = rest_of_content[pos_so_far:]
209 if remaining_content.strip():
210 blocks.append(('code', remaining_content, lineno))
211  
212 out = (file_conf, blocks)
213 if return_node:
214 out += (node,)
215 return out
